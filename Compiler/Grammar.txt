NumberLiteral : int * int option
DoubleQuotedString : string
Plus : unit
Minus : unit
Asterisk : unit
Slash : unit
ParenOpen : unit
ParenClose : unit

Expr -> ArithmeticSecondOrderExpr

ArithmeticSecondOrderExpr.Add -> ArithmeticSecondOrderExpr Plus ArithmeticFirstOrderExpr
ArithmeticSecondOrderExpr.Subtract -> ArithmeticSecondOrderExpr Minus ArithmeticFirstOrderExpr

ArithmeticFirstOrderExpr.Divide -> ArithmeticFirstOrderExpr Slash AtomExpr
ArithmeticFirstOrderExpr.Fallthrough -> AtomExpr

AtomExpr.Number -> NumberLiteral
AtomExpr.DoubleQuotedString -> DoubleQuotedString
