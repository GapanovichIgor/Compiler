NumberLiteral : int * int option
DoubleQuotedString : string
Identifier : string
Plus : unit
Minus : unit
Asterisk : unit
Slash : unit
ParenOpen : unit
ParenClose : unit
Let : unit
Equals : unit
Semicolon : unit
NewLine : unit
BlockOpen : unit
BlockClose : unit
InvalidToken : string

Program -> Expression

Expression.Binding -> Let Identifier Equals Expression BindingBody
Expression.Value -> ArithmeticSecondOrderExpression

ArithmeticSecondOrderExpression.Add -> ArithmeticSecondOrderExpression Plus ArithmeticFirstOrderExpression
ArithmeticSecondOrderExpression.Subtract -> ArithmeticSecondOrderExpression Minus ArithmeticFirstOrderExpression
ArithmeticSecondOrderExpression.Fallthrough -> ArithmeticFirstOrderExpression

ArithmeticFirstOrderExpression.Multiply -> ArithmeticFirstOrderExpression Asterisk Application
ArithmeticFirstOrderExpression.Divide -> ArithmeticFirstOrderExpression Slash Application
ArithmeticFirstOrderExpression.Fallthrough -> Application

Application.Application -> Application AtomExpression
Application.Fallthrough -> AtomExpression

AtomExpression.Number -> NumberLiteral
AtomExpression.DoubleQuotedString -> DoubleQuotedString
AtomExpression.Identifier -> Identifier
AtomExpression.Paren -> ParenOpening Expression ParenClosing
AtomExpression.Block -> BlockOpen Expression BlockClose

ParenOpening.Simple -> ParenOpen
ParenOpening.WithNewLine -> ParenOpen NewLine

ParenClosing.Simple -> ParenClose
ParenClosing.WithNewLine -> NewLine ParenClose

BindingBody.Semicolon -> Semicolon Expression
BindingBody.NewLine -> NewLine Expression
